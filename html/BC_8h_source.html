66  vector<double> AreaVectors, double SpecificHeatRatio)

70  double densityLive = LiveCellConservedVariables[0];
71  double uLive = LiveCellConservedVariables[1]/densityLive;
72  double vLive = LiveCellConservedVariables[2]/densityLive;
73  double wLive = LiveCellConservedVariables[3]/densityLive;
74  double pressureLive = (SpecificHeatRatio-1)*(LiveCellConservedVariables[4]-
75  0.5*densityLive*(uLive*uLive + vLive*vLive + wLive*wLive));
76 
77  double densityGhost = densityLive ;
78  double ughost = (1-2*n[0]*n[0])*uLive + (-2*n[0]*n[1])*vLive +
79  (-2*n[0]*n[2])*wLive;
80  double vghost = (-2*n[1]*n[0])*uLive + (1-2*n[1]*n[1])*vLive +
81  (-2*n[1]*n[2])*wLive;
82  double wghost = (-2*n[2]*n[0])*uLive + (-2*n[2]*n[1])*vLive +
83  (1-2*n[2]*n[2])*wLive;
84  double PressureGhost = pressureLive;
85 
86 
87  GhostCellConservedVariables[0] = densityGhost;
88  GhostCellConservedVariables[1] = densityGhost*ughost;
89  GhostCellConservedVariables[2] = densityGhost*vghost;
90  GhostCellConservedVariables[3] = densityGhost*wghost;
91  GhostCellConservedVariables[4] = (PressureGhost/(SpecificHeatRatio-1)) +
