57 double getMachDivergingDuct(double areaRatio)
58 {
59  double MachDiverging ;
60  //Diverging section
61  double mach_tolaerence = 1; // 0.001;
62  double MachDivergingLower = 1;
63  double MachDivergingUpper = 100;
64 
65  MachDiverging = 0.5*(MachDivergingLower + MachDivergingUpper);
69  double gamma = 1.4;
70  // calculating the diverging Mach first
71  while(mach_tolaerence > 0.001)
72  {
73  MachDiverging = 0.5*(MachDivergingUpper + MachDivergingLower);
74  double area_difference = areaRatio - pow((gamma+1)/2,-((gamma+1)/
75  (2*(gamma-1))))*pow((1+0.5*(gamma-1)*MachDiverging*MachDiverging),
76  ((gamma+1)/(2*(gamma-1))))/MachDiverging;
77 
78  if(area_difference<0)
79  {
80  MachDivergingUpper = MachDiverging;
81  }
82  else if(area_difference>=0)
83  {
84  MachDivergingLower = MachDiverging;
85  }
86  mach_tolaerence = MachDivergingUpper - MachDivergingLower;
87  }
88  return MachDiverging;
89 }
